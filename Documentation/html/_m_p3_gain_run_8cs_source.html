381 private void FindFiles(string parentFolder, BackgroundWorker searchWorker)
382 {
383 this.ContinueSearch = true;
384
385 searchWorker.ReportProgress(-1);
386
387
388 var folders = Directory.GetDirectories(parentFolder, "*", SearchOption.AllDirectories).ToList();
389 var songs = Directory.GetFiles(parentFolder, "*.MP3", SearchOption.AllDirectories).ToList();
390
391 searchWorker.ReportProgress(-1);
392
393 if (searchWorker.CancellationPending)
394 {
395 searchWorker.ReportProgress(100, CancelMessage);
396 return;
397 }
398
399 folders.Add(parentFolder);
400
401 var newFilesCount = this.GetNewFileCount(songs);
402
403 if (newFilesCount > 25000)
404 {
405 var question = $"Are you sure you want to continue loading {newFilesCount} songs?";
406
407 var result = MessageBox.Show(question, "MP3 Gain MT", MessageBoxButtons.YesNo, MessageBoxIcon.Question);
408
409 this.ContinueSearch = (result == DialogResult.Yes);
410
411 if (!this.ContinueSearch)
412 {
413 return;
414 }
415 }
416
417 var addedFilesCount = 0;
418
419 foreach (var folder in folders)
420 {
421 var mp3Folder = new MP3GainFolder(folder);
422
423 mp3Folder.FoundFile += MP3Folder_FoundFile;
424 mp3Folder.ChangedFile += MP3Folder_ChangedFile;
425 mp3Folder.SearchFolder();
426
427 if (mp3Folder.MP3Files.Count > 0)
428 {
429 if (!this.Folders.ContainsKey(folder))
430 {
431 this.Folders.Add(folder, mp3Folder);
432 addedFilesCount += mp3Folder.MP3Files.Count;
433
434 var progress = Helpers.GetProgress(addedFilesCount, newFilesCount);
435 searchWorker.ReportProgress(progress, mp3Folder);
436
437 if (searchWorker.CancellationPending)
438 {
439 searchWorker.ReportProgress(100, CancelMessage);
440 break;
441 }
442 }
443 }
444 }
445 }
